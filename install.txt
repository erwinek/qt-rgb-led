%YAML 1.1
---

define: &banner \r\nser2net port \p device \d [\B] (Debian GNU/Linux)\r\n\r\n

connection: &con01
    accepter: tcp,0.0.0.0,3333
    enable: on
    options:
      banner: *banner
      kickolduser: true
      telnet-brk-on-sync: true
      max-connections: 2               # do 2 klientów równocześnie

[Unit]
Description=LED Image Viewer
After=network.target

[Service]
Type=simple
ExecStart=/home/erwinek/matrix/led-image-viewer
